224Radical Radsqrt( int radicand ){
225 Radical res = {.out.up = 0, .out.down = 1, .in = 1};
226 if (0 == radicand) {
227 return res;
228 } else if (radicand < 0) {
229 fprintf(stderr, "Radsqrt: negative number, exiting\n");
230 exit(1);
231 }
232 int flag = 0;
233 for (int tmp = floor(sqrt(radicand)) ; tmp-1 ; tmp--) {
234 if (0 == radicand % (tmp * tmp)) {
235 res.out.up = tmp;
236 res.in = radicand / (tmp * tmp);
237 flag = 1;
238 break;
239 }
240 }
241 if (!flag) {
242 res.out.up = 1;
243 res.in = radicand;
244 }
245 return res;
