225 int lcmn = lcm_frac(a.down, b.down);

235Fraction mulFrac(Fraction a, Fraction b){
236 int gcdn1 = gcd_frac(a.up, b.down);
237 int gcdn2 = gcd_frac(a.down, b.up);
238 a.up /= gcdn1;
239 b.down /= gcdn1;
240 a.down /= gcdn2;
241 b.up /= gcdn2;
242 a.up *= b.up;
243 a.down *= b.down;
244 fixsignFrac(&a);
245 return a;
246}
